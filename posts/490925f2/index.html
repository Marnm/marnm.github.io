MySQL数据库高级应用之读写分离——MyCat篇

配置MySQL主从

Master

修改配置

symbolic-links=0

log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid

bind-address    = 0.0.0.0

user=mysql

innodb_flush_log_at_trx_commit=1
sync_binlog=1
log-bin=mysql-bin-master
server-id=1
binlog-do-db=ha
binlog-ignore-db=mysql
lower_case_table_names=1
skip-name-resolve

character-set-server=utf8
collation-server=utf8_general_ci

创建数据库，并设置授权用户

CREATE DATABASE ha;
use ha
CREATE TABLE test(id int, name char(16));

grant replication slave on *.* to root@'%' identified by '123456';

flush privileges;


show master status;

导出数据库

mysqldump -uroot -p -B ha > ha.sql

Slave

修改配置文件

symbolic-links=0

log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid

bind-address    = 0.0.0.0

log-bin=mysql-bin
server-id=3
binlog-do-db=ha
binlog-ignore-db=mysql
log_slave_updates=1
skip-name-resolve
lower_case_table_names=1

character-set-server=utf8
collation-server=utf8_general_ci

设置主服务

change master to master_host='master_ip_addr', master_user='root', master_password='123456', master_log_file='master_db_bin';
flush privileges;
start slave;

导入Master的数据库

mysql -uroot -p < ha.sql

安装配置MyCat

安装jdk，配置Java环境变量

mkdir /usr/java

tar -zxvf jdk-8u201-linux-x64.tar.gz -C /usr/java/

vim /etc/profile

JAVA_HOME=/usr/java/jdk1.8.0_201
PATH=$JAVA_HOME/bin:$PATH
CLASSPATH=$JAVA_HOME/jre/lib/ext:$JAVA_HOME/lib/tools.jar
export PATH JAVA_HOME CLASSPATH
MYCAT_HOME=/usr/local/mycat
PATH=$MYCAT_HOME/bin:$PATH

source /etc/profile

安装MyCat

tar -zxvf Mycat-server-1.6.7.3-release-20190927161129-linux.tar.gz -C /usr/local/
